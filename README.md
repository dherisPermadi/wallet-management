# README

# Requirement
  - PostgreSQL
  - Rails v. 7.0.4
  - Ruby v. 3.1.2
  - NodeJS
  - Yarn

# Install Gem / Libraries
  - run bundle install
  - install yarn
  - yarn add jquery
  - rails yarn:install

# Set the ENV Variable
  - Create .env file
  - Set value to variables HOST_DB, USERNAME_DB, PASSWORD_DB and PORT DB

# Database Build
  - run rake db:create
  - run rake db:migrate
  - run rake db:seed
